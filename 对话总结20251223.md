export interface KpiCardProps {
  title: string;
  value: string | number;
  hint?: string;
  variant?: 'default' | 'good' | 'warning' | 'danger';
  className?: string;
  onClick?: () => void;
}

export function KpiCard({ title, value, hint, variant = 'default', className, onClick }: KpiCardProps) {
  const isClickable = Boolean(onClick);
  
  return (
    <div
      className={cn(
        'rounded-xl border p-4 bg-white shadow-md transition-normal',
        isClickable && 'cursor-pointer hover:shadow-lg hover:-translate-y-0.5',
        variant === 'good' && 'border-status-good/20',
        variant === 'warning' && 'border-status-warning/20',
        variant === 'danger' && 'border-status-danger/20',
        className
      )}
      onClick={onClick}
      role={isClickable ? 'button' : undefined}
      tabIndex={isClickable ? 0 : undefined}
      onKeyDown={isClickable ? (e) => {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault();
          onClick?.();
        }
      } : undefined}
    >
      <div className="text-xs text-text-secondary">{title}</div>
      <div className={cn(
        'mt-2 text-lg font-semibold',
        variant === 'good' && 'text-status-good',
        variant === 'warning' && 'text-status-warning',
        variant === 'danger' && 'text-status-danger'
      )}>
        {value}
      </div>
      {hint && <div className="mt-2 text-xs text-text-muted">{hint}</div>}
    </div>
  );
}

export interface FilterSelectorProps<T extends string = string> {
  label: string;
  value: T;
  onChange: (value: T) => void;
  options: Array<{ value: T; label: string }>;
  active?: boolean;
  disabled?: boolean;
}

export function FilterSelector<T extends string = string>({
  label, value, onChange, options, active = false, disabled = false
}: FilterSelectorProps<T>) {
  return (
    <div className="flex flex-col gap-1">
      <label className="text-xs text-text-secondary">{label}</label>
      <select
        className={cn(
          'rounded-sm border px-2 py-1 text-sm bg-white/90',
          'transition-normal border-border-light',
          'hover:border-tesla hover:shadow-sm',
          active && 'border-tesla bg-tesla text-text-inverse',
          'focus:outline-none focus:shadow-focus',
          disabled && 'bg-border-light text-text-muted cursor-not-allowed'
        )}
        value={value}
        onChange={(e) => onChange(e.target.value as T)}
        disabled={disabled}
      >
        {options.map((opt) => (
          <option key={opt.value} value={opt.value}>{opt.label}</option>
        ))}
      </select>
    </div>
  );
}

export interface ChartContainerProps {
  title?: string;
  subtitle?: string;
  height?: number | 'sm' | 'md' | 'lg';
  children: React.ReactNode;
  className?: string;
}

export function ChartContainer({ title, subtitle, height = 'md', children, className }: ChartContainerProps) {
  const getHeight = (): string => {
    if (typeof height === 'number') return `${height}px`;
    const heightMap = { sm: '400px', md: '600px', lg: '600px' };
    return heightMap[height];
  };

  return (
    <div className={cn('w-full', className)}>
      {(title || subtitle) && (
        <div className="mb-4">
          {title && <h3 className="text-base font-semibold text-text-primary">{title}</h3>}
          {subtitle && <p className="mt-1 text-sm text-text-secondary">{subtitle}</p>}
        </div>
      )}
      <div className="w-full" style={{ height: getHeight() }}>{children}</div>

  );
